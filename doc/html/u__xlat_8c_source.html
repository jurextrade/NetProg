54XLAT_TBL* LdXlTbl(char* nomfic)
55{
56 int rc = 0;
57 int Etat = 0;
58 char From[MAXCODE+1], To[MAXCODE+1];
59 XLAT_TBL *new1, *sav;
60 FILE *f;
61 char buf[XLAT_MAXBUF+1];
62 int inverse = 0; /* = 1 si en modif, tables inversées */
63
64 if (( f = fopen( nomfic, "r" )) == NULL)
65 return NULL;
66
67 new1 = (XLAT_TBL *)malloc (sizeof(XLAT_TBL));
68
69 if (new1 == NULL)
70 {
71 fclose(f);
72 return NULL;
73 }
74
75 /* initialisation table */
76
77 /* lecture du fichier */
78
79 while (ReadSM(f, buf, XLAT_MAXBUF))
80 {
81 strtok(buf, ";*");
82 stripe(buf);
83 if (strpbrk(buf,";*") || (*buf == '\0'))
84 continue; /* sauter les lignes blanches ou comm */
85 /* strupr(buf);*/
86
87 switch (Etat)
88 {
89 case 0 :
90 case 1 :
91 if (strlen(buf) > 6)
92 {
93 rc = INVALID;
94 goto End;
95 }
96 switch (Etat)
97 {
98 case 0: /* lecture jeu départ */
99 strmcpy(From, buf, MAXCODE);
100 Etat = 1;
101 break;
102 case 1 : /* lecture jeu d'arrivée */
103 strmcpy(To, buf, MAXCODE);
104 /* vérification existence table */
105 if ((sav = GetXlat(To, From)) != NULL)
106 {
107 /* positionnement flag inversion */
108 inverse = strcmp(sav->From, From) != 0 ? 1 : 0;
109 /* passage en mode modif */
110 rc = ALREADY_EXISTS;
111 free (new1);
112 new1 = sav;
113 }
114 else
115 {
116 ListInsert (&TabXlat,new1);
117 strcpy(new1->From, From);
118 strcpy(new1->To, To);
119 }
120 Etat = 2;
121 break;
122 }
123 break;
124 case 2 : /* lecture des codes */
125 {
126 int n;
127 char c[2];
128 int i1, i2; /* codes départ et arrivée */
129
130 /* interprétation ligne */
131 n = sscanf(buf, "%2x %[<=>] %2x", &i1, c, &i2);
132 if (n != 3) break; /* ignorer ligne invalide */
133 switch (*c)
134 {
135 case '=' : /* conversion dans les deux tables */
136 new1->Tables[0][i1] = i2;
137 new1->Tables[1][i2] = i1;
138 break;
139 case '>' : /* conversion dans la table from -> to */
140 new1->Tables[inverse][i1] = i2;
141 break;
142 case '<' : /* conversion dans la table réciproque */
143 new1->Tables[not(inverse)][i2] = i1;
144 break;
145 }
146 }
147 } /* fin switch etat */
148 } /* fin while lecture */
149 /* positionnement flag bijective */
150 {
151 int i = 0;
152 new1->IsBiject = 1;
153 for(; i<256; i++)
154 if ((unsigned char)new1->Tables[1][(unsigned char)new1->Tables[0][i]] != i)
155 {
156 /* réciproque de l'image non égale ŕ élément départ */
157 new1->IsBiject = 0;
158 break;
159 }
160 }
161End :
162 ListInsert (&TabXlat, new1);
163 fclose( f );
164 return new1;
165}
