13int TPInit (TP* ptp, char* name, char* appliname)
14{
15 int i;
16 XF* pxf;
17
18 OS* pos = (OS*)malloc (sizeof (OS));
19 if (!pos)
20 {
21 OSError ("TEDAG001", NULL);
22 return -1;
23 }
24 if (OSInit (pos) < 0)
25 {
26 free (pos);
27 OSError ("TEDAG001", NULL);
28 return -1;
29 }
30 ptp->OS = pos;
31
32
33 /* SYNTAX */
34 pxf = (XF*)malloc (sizeof (XF));
35 if (!pxf)
36 {
37 OSError ("TEDAG001", NULL);
38 return -1;
39 }
40 if (XFInit (pxf, XFLittleEndian, XFAscii) < 0)
41 {
42 OSError ("TEDAG002", NULL);
43 }
44 ptp->XF = pxf;
45
46 /* BASECOM */
47 OSCLEARBITS(ptp->AllSockets);
48 OSCLEARBITS(ptp->AllClients);
49 OSCLEARBITS(ptp->AllFileClients);
50 OSCLEARBITS(ptp->AllBaseClients);
51 OSCLEARBITS(ptp->LastSelectMask);
52 OSCLEARBITS(ptp->ClientsWithInput);
53 OSCLEARBITS(ptp->ClientsWithOutput);
54 OSCLEARBITS(ptp->ClientsWithFileOutput);
55 OSCLEARBITS(ptp->ClientsWithFileInput);
56 OSCLEARBITS(ptp->ClientsWithBaseOutput);
57 OSCLEARBITS(ptp->ClientsWithBaseInput);
58 OSCLEARBITS(ptp->OutputPending);
59 OSCLEARBITS(ptp->InputPending);
60 OSCLEARBITS(ptp->SavedAllClients);
61 OSCLEARBITS(ptp->SavedAllSockets);
62 OSCLEARBITS(ptp->SavedClientsWithInput);
63 OSCLEARBITS(ptp->SavedClientsWithOutput);
64 OSCLEARBITS(ptp->SavedClientsWithFileInput);
65 OSCLEARBITS(ptp->SavedClientsWithFileOutput);
66 OSCLEARBITS(ptp->SavedClientsWithBaseInput);
67 OSCLEARBITS(ptp->SavedClientsWithBaseOutput);
68 OSCLEARBITS(ptp->ClientsWriteBlocked);
69 OSCLEARBITS(ptp->WellKnownConnections );
70 OSCLEARBITS(ptp->LocalListening );
71 OSCLEARBITS(ptp->DistantListening );
72 OSCLEARBITS(ptp->ConnectionPending );
73 ptp->CurrentMaxClients = ptp->NextFreeClientId = 0;
74 ptp->AnyClientsWriteBlocked = ptp->GrabInProgress = FALSE;
75 ptp->ApplicationProcedures = NULL;
76 ptp->ConnectionProcedures = NULL;
77 ptp->History = NULL;
78 ptp->ClientHistoryNumber = 0;
79
80 for (i = 0; i < SOCKMAX; i++)
81 {
82 ptp->ConnectionTranslation[i] = 0;
83 ptp->Clients[i] = NULL;
84 }
85 if (name)
86 strcpy (ptp->Name, name);
87 else strcpy (ptp->Name, "");
88
89 if (appliname)
90 strcpy (ptp->AppliName, appliname);
91 else strcpy (ptp->AppliName, "");
92
93 ptp->Mode = TPCLIENT;
94 ptp->ListeningPorts = NULL;
95 TPAddApplicationProcedure (ptp, TPSendIfAnyOutput, NULL);
96 TPAddApplicationProcedure (ptp, TPRecvIfAnyInput, NULL);
97 return 1;
98}

170void TPConnectionOK (TP* ptp, TPClient* pclient)
171{
172 int request = pclient->Pipe->fd;
173 DWORD one = 1;
174 List* plist;
175 TPConnectionProcedure* ConnectionProc;
176
177#ifdef SO_LINGER
178 int linger[2] = {0, 0};
179 OSSetSockOpt (ptp->OS, request, SOL_SOCKET, SO_LINGER, &linger, sizeof (linger));
180#endif
181 if (OSIoctl (ptp->OS, request, FIONBIO, &one) < 0)
182 OSError ("TESYS029", "socket : %d", request);
183
184 plist = ptp->ConnectionProcedures;
185 while (plist)
186 {
187 ConnectionProc = (TPConnectionProcedure *)plist->car;
188 ConnectionProc->ExecuteProcedure(pclient,ConnectionProc->ApplicationField);
189 plist = plist->cdr;
190 }
191 if ((pclient->Protocol != IPPROTO_ICMP) &&
192 (pclient->Protocol != IPPROTO_IP))
193 {
194 pclient->Connect = TRUE;
195 OSGetPeerName (ptp->OS, request, pclient->Type, pclient->WithUserId, pclient->WithName);
196 }
197}

672void TPCloseDownConnection (TP* ptp, TPClient* pclient)
673{
674 PIPE* pcom = pclient->Pipe;
675 int connection = pcom->fd;
676
677
678 if (pclient->Pipe->fileid)
679 {
680 OSCloseFile (pclient->Pipe->fileid);
681 OSFreePipe (pcom);
682 pclient->Pipe = NULL;
683 OSBITCLEAR(ptp->ClientsWithFileInput, pclient->Index + 1);
684 OSBITCLEAR(ptp->ClientsWithFileOutput, pclient->Index + 1);
685 OSBITCLEAR(ptp->AllFileClients, pclient->Index + 1);
686 if (ptp->GrabInProgress)
687 {
688 OSBITCLEAR(ptp->SavedClientsWithFileInput, pclient->Index + 1);
689 OSBITCLEAR(ptp->SavedClientsWithFileOutput, pclient->Index + 1);
690 }
691 return;
692 }
693 if (pclient->Pipe->connection)
694 {
695 OSDisconnect (pclient->Pipe->connection, pclient->Protocol, FALSE, NULL, NULL);
696 OSFreePipe (pcom);
697 pclient->Pipe = NULL;
698 OSBITCLEAR(ptp->ClientsWithBaseInput, pclient->Index + 1);
699 OSBITCLEAR(ptp->ClientsWithBaseOutput, pclient->Index + 1);
700 OSBITCLEAR(ptp->AllBaseClients, pclient->Index + 1);
701 if (ptp->GrabInProgress)
702 {
703 OSBITCLEAR(ptp->SavedClientsWithBaseInput, pclient->Index + 1);
704 OSBITCLEAR(ptp->SavedClientsWithBaseOutput, pclient->Index + 1);
705 }
706 return;
707 }
708 if (pcom->Output && pcom->Output->Count)
709 {
710 TPFlushClient (ptp, pclient);
711 }
712 OSCloseSocket (ptp->OS, connection);
713
714 if (pclient->OnPort)
715 {
716 ListRemove (&pclient->OnPort->Clients, pclient);
717 }
718
719 ptp->ConnectionTranslation[pcom->fd] = 0;
720
721 OSBITCLEAR(ptp->AllSockets, connection);
722 OSBITCLEAR(ptp->AllClients, connection);
723 OSBITCLEAR(ptp->ClientsWithInput, connection);
724 OSBITCLEAR(ptp->ClientsWithOutput, connection);
725 if (ptp->GrabInProgress)
726 {
727 OSBITCLEAR(ptp->SavedAllSockets, connection);
728 OSBITCLEAR(ptp->SavedAllClients, connection);
729 OSBITCLEAR(ptp->SavedClientsWithInput, connection);
730 OSBITCLEAR(ptp->SavedClientsWithOutput, connection);
731 }
732 OSBITCLEAR(ptp->ClientsWriteBlocked, connection);
733 if (!OSANYSET(ptp->ClientsWriteBlocked))
734 ptp->AnyClientsWriteBlocked = FALSE;
735 OSBITCLEAR(ptp->OutputPending, connection);
736 OSBITCLEAR(ptp->InputPending, connection);
737}

740int TPCheckConnection (TP* ptp, TPClient* pclient)
741{
742 struct timeval notime;
743 int ret;
744 long tmask[MaskCount];
745 int connection = pclient->Pipe->fd;
746
747
748 notime.tv_sec = 0;
749 notime.tv_usec = 0;
750 OSCLEARBITS(tmask);
751 OSBITSET(tmask, connection);
752
753 ret = OSSelect (ptp->OS, connection + 1, tmask, NULL, NULL, &notime);
754 if (ret < 0)
755 return TP_NOK;
756 else
757 return TP_OK;
758}

1188 waittime.tv_usec = microseconds;
1189 else
1190 {
1191 lt = ldiv (microseconds, 1000000);
1192 waittime.tv_usec = lt.rem;
1193 waittime.tv_sec = lt.quot;

1213 OSCLEARBITS(clientsWritable);

1234 OSMASKANDSETBITS (ptp->InputPending, ptp->LastSelectMask, ptp->AllClients);
1235
